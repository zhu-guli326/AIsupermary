        async function fetchWithTimeout(url, options, timeout = 10000) {
            const controller = new AbortController();
            const id = setTimeout(() => controller.abort(), timeout);
            const response = await fetch(url, {
                ...options,
                signal: controller.signal  
            });
            clearTimeout(id);
            return response;
        }
